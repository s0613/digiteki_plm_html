⑤ 사용 가이드
<!-- 기본 --> <div class="progress-digitek-linear" style="--percent: 65;"></div> <!-- 길이 + 색상 + 높이 커스텀 --> <div class="progress-digitek-linear" style="--percent: 50; --progress-width: 300px; --progress-height: 8px; --progress-color: #2F66BA; --progress-bg: rgba(47,102,186,0.16);"></div> <!-- Thymeleaf 동적 바인딩 --> <div class="progress-digitek-linear" th:style="'--percent:' + ${progress}"></div>
CSS 변수 기본값 설명
--percent 0 진행률 (0~100)
--progress-width 200px 바 전체 길이 (px, % 모두 가능)
--progress-height 6px 바 높이
--progress-color #29C23B 채워진 색상 (success 녹색)
--progress-bg rgba(41,194,59,0.16) 빈 배경 색상
